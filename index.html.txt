function uid() { return 'i' + Date.now() + Math.random().toString(36).slice(2,8); }
function yen(n) { return '¥' + Number(n).toLocaleString(); }
function fmtDate(s) {
  if (!s) return '';
  const [y,m,d] = s.split('-');
  return y + '年' + parseInt(m) + '月' + parseInt(d) + '日';
}
function getMonthStr(d) { return d.getFullYear() + '-' + String(d.getMonth()+1).padStart(2,'0'); }
function calcInv(inv) {
  let sub = 0;
  (inv.lines||[]).forEach(l => sub += (l.qty||0)*(l.price||0));
  const tax = Math.floor(sub * 0.1);
  return { sub, tax, total: sub+tax };
}

  document.querySelectorAll('.tab-content').forEach(t => t.classList.remove('active'));
  document.querySelectorAll('.tab-btn').forEach(b => b.classList.remove('active'));
  document.getElementById('tab-'+name).classList.add('active');
  btn.classList.add('active');
  if (name === 'preview') updatePreviewSelect();
  if (name === 'stores') renderStores();
}

// ===== INVOICE LIST =====
function loadInvoices() {
  const month = document.getElementById('currentMonth').value;
  const invs = data.invoices.filter(i => i.month === month);
  const tbody = document.getElementById('invoiceTableBody');

  if (!invs.length) {
    tbody.innerHTML = `<tr><td colspan="8" style="text-align:center;padding:40px;color:var(--ink-light)">
      本月暂无请求书，点击「⚡ 生成本月请求书」</td></tr>`;
    document.getElementById('totalRow').style.display = 'none';
    document.getElementById('monthStat').innerHTML = '';
    return;
  }

  let tSub=0, tTax=0, tTotal=0;
  tbody.innerHTML = invs.map(inv => {
    const {sub,tax,total} = calcInv(inv);
    tSub+=sub; tTax+=tax; tTotal+=total;
    const pl = (inv.lines||[]).find(l=>l.name.includes('技能'));
    return `<tr>
      <td><strong>${inv.name}</strong></td>
      <td style="font-size:11px;color:var(--ink-light)">${inv.periodStart?fmtDate(inv.periodStart)+'〜'+fmtDate(inv.periodEnd):''}</td>
      <td>${pl ? pl.qty+'人' : '-'}</td>
      <td>${yen(sub)}</td>
      <td>${yen(tax)}</td>
      <td class="amount">${yen(total)}</td>
      <td><span class="status-badge ${inv.status==='paid'?'status-paid':'status-pending'}">${inv.status==='paid'?'支払済':'未支払'}</span></td>
      <td class="actions">
        <button class="btn btn-outline btn-sm" onclick="editInvoice('${inv.id}')">編集</button>
        <button class="btn btn-pdf btn-sm" onclick="downloadOnePDF('${inv.id}')">📥PDF</button>
        <button class="btn btn-sm" style="background:#fee2e2;color:#991b1b" onclick="deleteInvoice('${inv.id}')">削除</button>
      </td>
    </tr>`;
  }).join('');

  document.getElementById('totalRow').style.display = 'flex';
  document.getElementById('totalSubtotal').textContent = yen(tSub);
  document.getElementById('totalTax').textContent = yen(tTax);
  document.getElementById('totalGrand').textContent = yen(tTotal);
  document.getElementById('monthStat').innerHTML = `共 <strong>${invs.length}</strong> 件　合計 <strong>${yen(tTotal)}</strong>`;
}

// ===== GENERATE =====
function generateMonth() {
  const month = document.getElementById('currentMonth').value;
  if (!month) { alert('请先选择月份'); return; }
  const existing = data.invoices.filter(i=>i.month===month);
  if (existing.length && !confirm(`本月已有 ${existing.length} 件，是否覆盖重新生成？`)) return;
  data.invoices = data.invoices.filter(i=>i.month!==month);

  const [y,m] = month.split('-').map(Number);
  const ld = lastDayOf(y,m);
  const ps = `${y}-${String(m).padStart(2,'0')}-01`;
  const pe = `${y}-${String(m).padStart(2,'0')}-${ld}`;
  const due = `${m===12?y+1:y}-${String(m===12?1:m+1).padStart(2,'0')}-28`;

  data.stores.forEach(s => {
    const lines = [{ name:`${y}.${m}.1-${y}.${m}.${ld} 月額料金`, qty:1, price:s.monthly }];
    if (s.persons>0) lines.push({ name:'特定技能支援費', qty:s.persons, price:10000 });
    data.invoices.push({ id:uid(), month, name:s.name, storeId:s.id, periodStart:ps, periodEnd:pe, due, lines, status:'pending' });
  });
  saveData(); loadInvoices();
  alert(`✅ 已生成 ${data.stores.length} 件请求书！`);
}

// ===== PDF GENERATION =====
function showLoading(msg) {
  document.getElementById('pdfLoadingMsg').textContent = msg || '正在生成PDF...';
  document.getElementById('pdfLoading').classList.add('show');
}
function hideLoading() { document.getElementById('pdfLoading').classList.remove('show'); }

async function renderPageToPDF(jspdf, inv, pageIndex) {
  const area = document.getElementById('pdfRenderArea');
  area.innerHTML = buildInvHTML(inv);
  await new Promise(r => setTimeout(r, 300));

  const canvas = await html2canvas(area.firstChild, {
    scale: 2, useCORS: true, backgroundColor: '#ffffff',
    width: 794, windowWidth: 794
  });

  const imgData = canvas.toDataURL('image/jpeg', 0.95);
  if (pageIndex > 0) jspdf.addPage();
  const pdfW = jspdf.internal.pageSize.getWidth();
  const pdfH = jspdf.internal.pageSize.getHeight();
  jspdf.addImage(imgData, 'JPEG', 0, 0, pdfW, pdfH);
}

async function downloadOnePDF(id) {
  const inv = data.invoices.find(i=>i.id===id);
  if (!inv) return;
  showLoading(`正在生成：${inv.name}`);
  try {
    const { jsPDF } = window.jspdf;
    const doc = new jsPDF({ orientation:'portrait', unit:'mm', format:'a4' });
    await renderPageToPDF(doc, inv, 0);
    const safeName = inv.name.replace(/[\/\\:*?"<>|]/g,'_');
    doc.save(`請求書_${safeName}_${inv.month}.pdf`);
  } catch(e) {
    alert('PDF生成失败：' + e.message);
  } finally {
    document.getElementById('pdfRenderArea').innerHTML = '';
    hideLoading();
  }
}

async function downloadAllPDF() {
  const month = document.getElementById('currentMonth').value;
  const invs = data.invoices.filter(i=>i.month===month);
  if (!invs.length) { alert('本月暂无请求书'); return; }
  showLoading(`正在生成 ${invs.length} 张请求书... (1/${invs.length})`);
  try {
    const { jsPDF } = window.jspdf;
    const doc = new jsPDF({ orientation:'portrait', unit:'mm', format:'a4' });
    for (let i=0; i<invs.length; i++) {
      document.getElementById('pdfLoadingMsg').textContent = `正在处理：${invs[i].name} (${i+1}/${invs.length})`;
      await renderPageToPDF(doc, invs[i], i);
    }
    doc.save(`請求書_${month}_全店舗.pdf`);
  } catch(e) {
    alert('PDF生成失败：' + e.message);
  } finally {
    document.getElementById('pdfRenderArea').innerHTML = '';
    hideLoading();
  }
}

function onStoreSelect() {
  const sid = document.getElementById('inv-store').value;
  if (!sid) return;
  const s = data.stores.find(x=>x.id===sid);
  if (!s) return;
  document.getElementById('inv-name').value = s.name;
  const month = document.getElementById('currentMonth').value;
  if (month) {
    const [y,m] = month.split('-').map(Number);
    const ld = lastDayOf(y,m);
    const lines = [{name:`${y}.${m}.1-${y}.${m}.${ld} 月額料金`,qty:1,price:s.monthly}];
    if (s.persons>0) lines.push({name:'特定技能支援費',qty:s.persons,price:10000});
    setLineItems(lines);
  }
}

function setLineItems(lines) {
  document.getElementById('lineItemsBody').innerHTML = '';
  lines.forEach(l => addLineItem(l));
  updateModalCalc();
}

function addLineItem(line) {
  const body = document.getElementById('lineItemsBody');
  const row = document.createElement('div');
  row.className = 'line-item-row';
  row.innerHTML = `
    <input type="text" placeholder="项目名" value="${(line&&line.name)||''}" oninput="updateModalCalc()">
    <input type="number" placeholder="数量" value="${(line&&line.qty)||1}" min="0" oninput="updateModalCalc()">
    <input type="number" placeholder="単価" value="${(line&&line.price)||0}" min="0" oninput="updateModalCalc()">
    <span class="line-item-total">¥0</span>
    <button class="btn-icon" onclick="this.closest('.line-item-row').remove();updateModalCalc()">×</button>`;
  body.appendChild(row);
  updateModalCalc();
}

function updateModalCalc() {
  let sub = 0;
  document.querySelectorAll('.line-item-row').forEach(row => {
    const inp = row.querySelectorAll('input');
    const amt = (parseFloat(inp[1].value)||0)*(parseFloat(inp[2].value)||0);
    sub += amt;
    row.querySelector('.line-item-total').textContent = yen(amt);
  });
  const tax = Math.floor(sub*0.1);
  document.getElementById('modalSubtotal').textContent = yen(sub);
  document.getElementById('modalTax').textContent = yen(tax);
  document.getElementById('modalTotal').textContent = yen(sub+tax);
}

function saveInvoice() {
  const name = document.getElementById('inv-name').value.trim();
  if (!name) { alert('请输入请求对象名称'); return; }
  const lines = [];
  document.querySelectorAll('.line-item-row').forEach(row => {
    const inp = row.querySelectorAll('input');
    lines.push({ name:inp[0].value, qty:parseFloat(inp[1].value)||0, price:parseFloat(inp[2].value)||0 });
  });
  const inv = {
    id: editingInvoiceId || uid(),
    month: document.getElementById('currentMonth').value,
    name, storeId: document.getElementById('inv-store').value,
    periodStart: document.getElementById('inv-period-start').value,
    periodEnd: document.getElementById('inv-period-end').value,
    due: document.getElementById('inv-due').value,
    lines, status: document.getElementById('inv-status').value
  };
  if (editingInvoiceId) {
    const idx = data.invoices.findIndex(i=>i.id===editingInvoiceId);
    data.invoices[idx] = inv;
  } else {
    data.invoices.push(inv);
  }
  saveData(); closeModal('invoiceModal'); loadInvoices();
}

function deleteInvoice(id) {
  if (!confirm('确认删除？')) return;
  data.invoices = data.invoices.filter(i=>i.id!==id);
  saveData(); loadInvoices();
}

function editStore(id) {
  const s = data.stores.find(x=>x.id===id);
  if (!s) return;
  editingStoreId = id;
  document.getElementById('storeModalTitle').textContent = '编辑店铺';

function saveStore() {
  const name = document.getElementById('st-name').value.trim();
  if (!name) { alert('请输入店铺名称'); return; }
  const store = { id:editingStoreId||uid(), name, monthly:parseFloat(document.getElementById('st-monthly').value)||9000, persons:parseInt(document.getElementById('st-persons').value)||0, note:document.getElementById('st-note').value.trim() };
  if (editingStoreId) { const idx=data.stores.findIndex(s=>s.id===editingStoreId); data.stores[idx]=store; }
  else data.stores.push(store);
  saveData(); closeModal('storeModal'); renderStores();
}

function deleteStore(id) {
  if (!confirm('确认删除？')) return;
  data.stores = data.stores.filter(s=>s.id!==id);
  saveData(); renderStores();
}

// ===== AUTO GENERATE =====
function checkAutoGen() {
  const today = new Date();
  if (today.getDate()!==1) return;
  const thisMonth = getMonthStr(today);
  if (data.lastAutoGenMonth===thisMonth) return;
  if (data.invoices.filter(i=>i.month===thisMonth).length) { data.lastAutoGenMonth=thisMonth; saveData(); return; }
  document.getElementById('currentMonth').value = thisMonth;
  generateMonth();
  data.lastAutoGenMonth = thisMonth;
  saveData();
}
